cd  -> Nos movemos a una carpeta
git init  ->Iniciar git repositorio
git status  -> Ver cuales ya se subieron y cuales no
git add (nombre del archivo)    ->Para decir cual subir 
git add -A  ->Subir todo lo de la carpeta
git commit -m "Aqui se escriben los comentarios"
